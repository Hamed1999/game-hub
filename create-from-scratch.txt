# Create Vite-React-Typescript Project and init git repository
npm create vite@4.1.0 -> 7.1.9
npm install
git init
git add .
git commit -m "Initial Commit"
--------------------------------------------------------------------------------------------------------------------------
# Instal Chakra UI library as instructions on https://chakra-ui.com/docs/get-started/frameworks/vite
npm i @chakra-ui/react @emotion/react
npx @chakra-ui/cli snippet add
npm i -D vite-tsconfig-paths
--------------------------------------------------------------------------------------------------------------------------
# Install axios to send http requests
npm i axios
--------------------------------------------------------------------------------------------------------------------------
# Instal React icons 
npm i react-icons@4.7.1
--------------------------------------------------------------------------------------------------------------------------
# To build the application
npm run build
npm i -g vercel
